Line Charts

Line charts take two or three parameters.

  • First parameter - X-axis is time field or numeric expression

  • Second parameter - Y-axis is a numeric expression

  • Third (optional) Pivot is dimensional field (numeric or string)

Data Style is 'line_chart'

WITH __stage0 AS (
  SELECT
    group_set,
    CASE WHEN group_set=1 THEN DATE_TRUNC(DATE(base.dep_time, 'UTC'), MONTH) END as departure_month__1,
    CASE WHEN group_set=1 THEN COUNT( 1) END as flight_count__1
  FROM `malloy-data.faa.flights` as base
  CROSS JOIN (SELECT row_number() OVER() -1  group_set FROM UNNEST(GENERATE_ARRAY(0,1,1)))
  GROUP BY 1,2
)
SELECT
  ARRAY_AGG(CASE WHEN group_set=1 THEN STRUCT(departure_month__1 as departure_month, flight_count__1 as flight_count) END IGNORE NULLS  ORDER BY  departure_month__1 desc ) as departures_by_month
FROM __stage0
