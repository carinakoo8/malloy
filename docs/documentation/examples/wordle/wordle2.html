Understanding Letter Frequency

Starting with the query we built in step one, we built a query that produces a table of words with subtable where each row is the letter and position in that row.

Create a new explore wordle to query the data in this form. 

explore: wordle is from(-> words_and_position) {
  measure: word_count is count()
}

WITH __stage0 AS (
  SELECT 
     upper(words.word) as word
  FROM `malloy-data.malloytest.words_bigger` as words
  WHERE ((length(words.word))=5)and(REGEXP_CONTAINS(words.word,r'^[a-z]{5}$'))
)
, __stage1 AS (
  SELECT
    group_set,
    base.word as word__0,
    CASE WHEN group_set=1 THEN substr(base.word,numbers_0.num,1) END as letter__1,
    CASE WHEN group_set=1 THEN numbers_0.num END as position__1
  FROM __stage0 as base
  LEFT JOIN `malloy-data.malloytest.numbers` AS numbers_0 ON 1=1 AND numbers_0.num<=5
  CROSS JOIN (SELECT row_number() OVER() -1  group_set FROM UNNEST(GENERATE_ARRAY(0,1,1)))
  GROUP BY 1,2,3,4
)
, __stage2 AS (
  SELECT
    word__0 as word,
    ARRAY_AGG(CASE WHEN group_set=1 THEN STRUCT(letter__1 as letter, position__1 as position) END IGNORE NULLS  ORDER BY  position__1 ASC ) as letters
  FROM __stage1
  GROUP BY 1
  ORDER BY 1 asc
)
SELECT 
   COUNT( 1) as word_count
FROM __stage2 as wordle
ORDER BY 1 desc

  SELECT 
     upper(words.word) as word
  FROM `malloy-data.malloytest.words_bigger` as words
  WHERE ((length(words.word))=5)and(REGEXP_CONTAINS(words.word,r'^[a-z]{5}$'))
)
, __stage1 AS (
  SELECT
    group_set,
    base.word as word__0,
    CASE WHEN group_set=1 THEN substr(base.word,numbers_0.num,1) END as letter__1,
    CASE WHEN group_set=1 THEN numbers_0.num END as position__1
  FROM __stage0 as base
  LEFT JOIN `malloy-data.malloytest.numbers` AS numbers_0 ON 1=1 AND numbers_0.num<=5
  CROSS JOIN (SELECT row_number() OVER() -1  group_set FROM UNNEST(GENERATE_ARRAY(0,1,1)))
  GROUP BY 1,2,3,4
)
, __stage2 AS (
  SELECT
    word__0 as word,
    ARRAY_AGG(CASE WHEN group_set=1 THEN STRUCT(letter__1 as letter, position__1 as position) END IGNORE NULLS  ORDER BY  position__1 ASC ) as letters
  FROM __stage1
  GROUP BY 1
  ORDER BY 1 asc
)
, __stage3 AS (
  SELECT
    group_set,
    letters_0.letter as letter__0,
    CASE WHEN group_set=0 THEN COUNT(DISTINCT wordle.word) END as word_count__0,
    CASE WHEN group_set=0 THEN COUNT( 1) END as use_count__0,
    CASE WHEN group_set=1 THEN letters_0.position END as position__1,
    CASE WHEN group_set=1 THEN COUNT(DISTINCT wordle.word) END as word_count__1
  FROM __stage2 as wordle
  , UNNEST(wordle.letters) as letters_0
  CROSS JOIN (SELECT row_number() OVER() -1  group_set FROM UNNEST(GENERATE_ARRAY(0,1,1)))
  GROUP BY 1,2,5
)
SELECT
  letter__0 as letter,
  ANY_VALUE(CASE WHEN group_set=0 THEN word_count__0 END) as word_count,
  ANY_VALUE(CASE WHEN group_set=0 THEN use_count__0 END) as use_count,
  ARRAY_AGG(CASE WHEN group_set=1 THEN STRUCT(position__1 as position, word_count__1 as word_count) END IGNORE NULLS  ORDER BY  word_count__1 desc ) as positition_order_bar_chart
FROM __stage3
GROUP BY 1
ORDER BY 2 desc

Removing Plurals and words that end in 'ED'

We've noticed there are a lots of words that end in 'S' or 'ED' in the dataset, but in our experience they don't often appear in puzzles. We've eliminated them from our model for now, by filtering them out on the explore level:

explore: wordle is from(-> words_and_position) {
  where: word !~ r'(S|ED)$'
  measure: word_count is count()
}

, __stage1 AS (
  SELECT
    group_set,
    base.word as word__0,
    CASE WHEN group_set=1 THEN substr(base.word,numbers_0.num,1) END as letter__1,
    CASE WHEN group_set=1 THEN numbers_0.num END as position__1
  FROM __stage0 as base
  LEFT JOIN `malloy-data.malloytest.numbers` AS numbers_0 ON 1=1 AND numbers_0.num<=5
  CROSS JOIN (SELECT row_number() OVER() -1  group_set FROM UNNEST(GENERATE_ARRAY(0,1,1)))
  GROUP BY 1,2,3,4
)
, __stage2 AS (
  SELECT
    word__0 as word,
    ARRAY_AGG(CASE WHEN group_set=1 THEN STRUCT(letter__1 as letter, position__1 as position) END IGNORE NULLS  ORDER BY  position__1 ASC ) as letters
  FROM __stage1
  GROUP BY 1
  ORDER BY 1 asc
)
SELECT 
   COUNT( 1) as word_count
FROM __stage2 as wordle
WHERE IFNULL(NOT(REGEXP_CONTAINS(wordle.word,r'(S|ED)$')),FALSE)
ORDER BY 1 desc
