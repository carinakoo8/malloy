Google Analytics

You can find the complete source code for this model here.

Start by defining a source based on a query.

source: ga_sesions is table('bigquery-public-data.google_analytics_sample.ga_sessions_20170801') {
  dimension: start_time is timestamp_seconds(visitStartTime)
  measure:
    user_count is count(distinct fullVisitorId)
    session_count is count()
    total_visits is totals.visits.sum()
    total_hits is totals.hits.sum()
    total_page_views is totals.pageviews.sum()
    total_productRevenue is hits.product.productRevenue.sum()
    sold_count is hits.count() { where: hits.product.productQuantity > 0 }
}

We can then add a few named queries to the model to easily access or reference elsewhere.

  query: by_region is {
    top: 10
    where: geoNetwork.region !~ '%demo%'
    group_by: geoNetwork.region
    aggregate: user_count
  }

  query: by_device is {
    group_by: device.browser
    aggregate: user_count
    group_by: device.deviceCategory
  }

  query: by_source is {
    top: 10
    where: trafficSource.source != '(direct)'
    group_by: trafficSource.source
    aggregate: hits_count is hits.count()
  }

Putting it all together 

query: sessions_dashboard is ga_sessions -> {
  nest:
    by_region
    by_device
    by_source
    by_category is {
      group_by: category is hits.product.v2ProductCategory
      aggregate: total_productRevenue, sold_count
    }
}

WITH __stage0 AS (
  SELECT
    group_set,
    CASE WHEN group_set=1 THEN
      ga_sessions.geoNetwork.region
      END as region__1,
    CASE WHEN group_set=1 THEN
      count(distinct ga_sessions.fullVisitorId)
      END as user_count__1,
    CASE WHEN group_set=2 THEN
      ga_sessions.device.browser
      END as browser__2,
    CASE WHEN group_set=2 THEN
      count(distinct ga_sessions.fullVisitorId)
      END as user_count__2,
    CASE WHEN group_set=2 THEN
      ga_sessions.device.deviceCategory
      END as deviceCategory__2,
    CASE WHEN group_set=3 THEN
      ga_sessions.trafficSource.source
      END as source__3,
    CASE WHEN group_set=3 THEN
      COUNT(DISTINCT CONCAT(ga_sessions.__distinct_key, 'x', hits_0.__row_id))
      END as hits_count__3,
    CASE WHEN group_set=4 THEN
      product_0.v2ProductCategory
      END as category__4,
    CASE WHEN group_set=4 THEN
      COALESCE(SUM(product_0.productRevenue),0)
      END as total_productRevenue__4,
    CASE WHEN group_set=4 THEN
      COUNT(DISTINCT CASE WHEN product_0.productQuantity>0 THEN CONCAT(ga_sessions.__distinct_key, 'x', hits_0.__row_id) END)
      END as sold_count__4
  FROM (SELECT GENERATE_UUID() as __distinct_key, * FROM `bigquery-public-data.google_analytics_sample.ga_sessions_20170801` as x) as ga_sessions
  LEFT JOIN UNNEST(ARRAY(( SELECT AS STRUCT row_number() over() as __row_id, * FROM UNNEST(ga_sessions.hits)))) as hits_0
  LEFT JOIN UNNEST(hits_0.product) as product_0
  CROSS JOIN (SELECT row_number() OVER() -1  group_set FROM UNNEST(GENERATE_ARRAY(0,4,1)))
  WHERE ((group_set NOT IN (1) OR (group_set IN (1) AND COALESCE(NOT(ga_sessions.geoNetwork.region LIKE '%demo%'),FALSE))))
  AND ((group_set NOT IN (3) OR (group_set IN (3) AND COALESCE(NOT(ga_sessions.trafficSource.source='(direct)'),FALSE))))
  GROUP BY 1,2,4,6,7,9
)
SELECT
  ARRAY_AGG(CASE WHEN group_set=1 THEN STRUCT(
    region__1 as region, 
    user_count__1 as user_count
    ) END IGNORE NULLS  ORDER BY  user_count__1 desc  LIMIT 10) as by_region,
  ARRAY_AGG(CASE WHEN group_set=2 THEN STRUCT(
    browser__2 as browser, 
    user_count__2 as user_count, 
    deviceCategory__2 as deviceCategory
    ) END IGNORE NULLS  ORDER BY  user_count__2 desc ) as by_device,
  ARRAY_AGG(CASE WHEN group_set=3 THEN STRUCT(
    source__3 as source, 
    hits_count__3 as hits_count
    ) END IGNORE NULLS  ORDER BY  hits_count__3 desc  LIMIT 10) as by_source,
  ARRAY_AGG(CASE WHEN group_set=4 THEN STRUCT(
    category__4 as category, 
    total_productRevenue__4 as total_productRevenue, 
    sold_count__4 as sold_count
    ) END IGNORE NULLS  ORDER BY  total_productRevenue__4 desc ) as by_category
FROM __stage0
